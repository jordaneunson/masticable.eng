#!/bin/gastro

dutchOven(mediumHeat) {
    add: _baconStrips
    fry: tillCrispy
    add: _onion(1){blitzed}
    add: _celery(1){blitzed}
    add: _leek(1){chopped}
    # leek is optional
    cook: tillSoft
}

dutchOven(highHeat) {
    add: _russetPotatoes(1lbs){cubed}
    add: _chickenStock(to cover)
    add: _bayLeaf(2)
    boil
}

dutchOven(lowHeat) {
    add: _heavyCream(250ml)
    simmer
}

garnish: _cheddarCheese, _chives
