#!/bin/gastro

dutchOven(mediumHeat) {
    add: _bacon(3strips){chopped}
    fry: tillDesired
    add: _onion(1){diced}
    add: _celery(1){sliced}
    add: _carrot(1){sliced}
    cook: tillSoft
}

dutchOven(highHeat) {
    add: _newPotatoes(enough){diced}
    add: _chickenBroth(to cover)
    add: _bayLeaf(2)
    add: _thyme
    boil
}

smallPot(lowHeat) {
    add: _milk(500ml)
    add: _butter(2tbsp)
    add: _flour(2tbsp)
    stir: untilThickened
}

dutchOven(noHeat) {
    add: smallPot()
    add: _cannedClams{withJuice}
    add: _dill(the whole god damn package){chopped finely}
}

serve
