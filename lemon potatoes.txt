#!/bin/gastro

oven(390F) {
    preheat
}

largePot(highHeat) {
    add: _water
    boil
    add: _russetPotatoes(1kg){peeled, quartered}
    add: _salt
    add: _bakingSoda(1tsp)
    wait: 2mins
    drain
}

largeBowl() {
    _garlic(6cloves){crushed}
    _yellowMustard(2tbsp)
    _lemonJuice(1/3cup)
    _oregano(1tbsp)
    _oliveOil(1/2cup)
    _salt(1tsp)
    _pepper(1tsp)
}

largeBowl() {
    add: _russetPotatoes
    toss
}

largeZipLock() {
    add: largeBowl()
    fridge: 1hr min
}

largeCasseroleDish(oven, 390F) {
    add: largeZipLock()
    # place a baking sheet over top
    bake: 1hr
    # remove baking sheet
    bake: 45mins
}

serve
