#!/bin/gastro

oven(350F) {
    preheat
}

mediumPot(highHeat) {
    add: _macaroni
    cook: 85%
    # pull early, it'll keep cooking in the oven
    drain
}

smallPot(mediumHeat) {
    add: _butter(1/2cup)
    add: _flour(1/2cup)
    stir: 15mins
    # this is the roux
    add: _cheese(1/2cup){variousTypes}
    add: _garlic(2cloves){crushed}
    add: _rosemary
    add: _oregano
    stir: untilMelted
}

casseroleDish() {
    add: mediumPot()
    add: smallPot()
    mix
    top: _cheese(lots){shredded}
}

oven(350F) {
    bake: tillGolden
}

serve
