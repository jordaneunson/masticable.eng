#!/bin/jordan

oven(350F) {}
mediumPot(highHeat) {
	_macaroni(pull at 85%)
}

smallPot(mediumHeat) {
	roux() {
		_butter(1/2cup)
		_flour(1/2cup)
		stir(){15mins}
	}
	_cheese(1/2cup){variousTypes}
	flavourNugget(crushedandmixed) {
		_garlic(2cloves)
		_rosemary
		_oregano
	}
}

casseroleDish() {
	mediumPot()
	smallPot()
	mix()
	_cheese(lots){shredded}
}
	
bake()
eat()
