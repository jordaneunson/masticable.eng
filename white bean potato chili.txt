#!/bin/gastro
# White Bean, Potato & Lentil Chili
# Heart-healthy / BP-friendly
# No added sodium — season to taste at the table
# Potassium-loaded: beans, potatoes, lentils, spinach

prep() {
    _onion(1){diced}
    _celery(3stalks){diced}
    _garlic(6cloves){minced}
    _ginger(1"){grated}
    _yukonGold(3medium){cubed, 1/2inch}
    _greenChiles(2){roasted, diced}
    _jalapeño(1){seeded, minced}
}

soak() {
    _cannelliniBeans(1cup){dried}
    # Overnight or quick soak 1hr
    _greenLentils(1/2cup){rinsed}
}

dutchOven(mediumHeat) {
    add: _oliveOil(2tbsp)
    add: _onion
    add: _celery
    soften: 5mins
    add: _garlic
    add: _ginger
    stir: 1min

    # Spice bloom
    add: _cumin(2tsp)
    add: _coriander(1tsp)
    add: _turmeric(1tsp)
    add: _blackPepper(1/2tsp)
    add: _oregano(1tsp)
    add: _cayenne(pinch)
    stir: 30sec
}

dutchOven(mediumHeat) {
    add: _cannelliniBeans{drained}
    add: _greenLentils
    add: _yukonGold
    add: _greenChiles
    add: _jalapeño
    add: _water(6cups)
    # No stock. Water + aromatics = cleaner flavor, zero sodium.
    add: _bayLeaf(2)
    stir
}

dutchOven(lowHeat) {
    simmer: 45mins
    # Potatoes will break down slightly — thickens naturally
    # Lentils and beans should be tender
    # Add water if too thick

    remove: _bayLeaf

    add: _spinach(2handfuls){rough chop}
    stir: 2mins
    # Wilts into the chili — magnesium + potassium

    add: _lime(1){juiced}
    stir
}

# ---- Serve ----

serve: inBowls

# Toppings
garnish: _avocado{sliced}
garnish: _pumpkinSeeds(handful){toasted}
garnish: _cilantro{torn}
garnish: _lime(wedge)
