#!/bin/gastro
# Servings: 2
# Time: 30mins
# Difficulty: easy
# DASH: plant-based, high fiber, no added salt

mediumPot(highHeat, boilingWater) {
    add: _farro(1cup)
    # Farro: ~25mins, nutty and chewy — holds up well under toppings
    simmer: 25mins{untilTender}
    drain
}

prepCounter(noHeat) {
    _cannedBlackBeans(15oz){drained, rinsed, noSaltAdded}
    _avocado(1){sliced}
    _salsa(generous){noSaltAdded}
    _lime(1){cut into wedges}
    _freshCilantro{chopped}
    _redCabbage{shredded, optional}
}

bowl {
    layer: _farro
    add: _blackBeans
    add: _avocado
    add: _salsa
    garnish: _cilantro, _limeWedge
    # Squeeze lime over everything before eating
}
