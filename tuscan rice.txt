#!/bin/gastro

prepCounter() {
    _sausages(1 per person)
    remove: skin
}

dutchOven(mediumHeat) {
    add: _oliveOil(1tsp)
    add: _redOnion{chopped fine}
    add: _redPepper{diced}
    add: _garlicClove(2)
    cook: tillSoft
}

dutchOven(highHeat) {
    # make a well in the middle
    add: _sausages{crumbled}
    brown
    add: _paprika
    add: _rice
    fry: 30seconds
    add: _whiteWine
    add: _broth
}

dutchOven(lowHeat) {
    simmer: 12-15mins
}

dutchOven(noHeat) {
    add: _greenOnions
    add: _tomatoes{chopped, no seeds}
    add: _flatLeafParsley
}

serve
