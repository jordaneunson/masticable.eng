#!/bin/gastro
# Notes:
# - Improve rice.
# - Heavy cream: 250ml too small.
# - Honey is important for final flavor.
# - BBQ the chicken on a stick—better.

paste() {
    _garlic(4)
    _ginger(1")
    _greenChili(0.5)
}

marinade() {
    add: paste()
    add: _yogurt(4tbsp)
    add: _flour(1tbsp)
    mix
    add: _coriander(1tsp)
    add: _cumin(1tsp)
    add: _cinnamon(pinch)
    add: _redChili(2tsp)
    add: _saffron(pinch)
}

prepCounter() {
    _chickenBreasts(2){sliced}
    coat: marinade()
}

fridge() {
    wait: 2hrs
}

# ---- 2 Hours Later ----

bbq(highHeat) {
    cook: _chicken
    # Turn often, want charring. Don't burn.
}

gravyPot(mediumLowHeat) {
    add: _butter(cube)
    melt
    add: _fenugreek(25seeds)
    # _driedFenugreekLeaves are better
    add: _greenCardamom(6pods)
    add: _cloves(5)
    add: _cinnamonStick(piece)
    add: _ginger(1/2inch)
    add: _chili(1/2)
    wait: 10mins
    add: _onion(1/2)
    # Do not brown.
    add: _corianderPowder(1tsp)
    add: _chiliPowder(1tsp){heaping}
    add: _tomatoPaste(3heaps)
    add: _doubleCream
    add: _honey
    add: _freshCorianderStalks
    add: _salt(pinch)
    add: _lemon(squeeze)
    simmer: 3/10
}

gravyPot(lowHeat) {
    add: _chicken
}

serve: overRice
