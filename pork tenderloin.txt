#!/bin/gastro

oven(375F) {
    preheat
}

largeZipLock(fridge) {
    add: _porkTenderloin(0.5kg)
    add: _italianSaladDressing(some)
    wait: 12hrs
}

# ---- 12 Hours Later ----

herbMix() {
    _garlic(6cloves){crushed}
    _basil(lots){diced}
    _oregano(lots){diced}
    _thyme(lots){NO STEMS}
    _sage(lots){diced}
    _parsley(lots){diced}
}

aluminumFoil(longSheet) {
    add: _porkTenderloin
    add: _seasoning(lots)
    add: herbMix()
    add: _butter(pats){placed on top}
    # wrap the meat like a burrito in the foil
}

oven(375F) {
    add: aluminumFoil()
    while (internalTemp < 145F) {
        wait
    }
}

rest: 5mins

serve
