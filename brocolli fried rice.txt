#!/bin/gastro

soyMix() {
    _soySauce(1tbsp)
    _ponzu(1tbsp)
    _sugar(1tbsp)
    _ginger(1inch){grated}
    _garlic(3cloves){crushed}
}

veg() {
    _broccoliHead{florets, stem}
    _scallions
}

fryPan(highHeat) {
    add: _oil
    add: _dayOldRice(1cup)
    fry: tillCrispy
    add: _eggs(3){whisked}
    add: soyMix()
    add: veg()
    add: _sesameOil(1tsp)
}

serve
