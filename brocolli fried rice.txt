#!/bin/broccoli_fried_rice

soyMix() {
    _soySauce(1tbsp)
    _ponzu(1tbsp)
    _sugar(1tbsp)
    _ginger(1inch, grated)
    _garlic(3cloves, crushed)
}

veg() {
    _broccoliHead(florets, stem)
    _scallions
}

fryPan(highHeat, oil) {
    _dayOldRice(1cup)
}

if (_dayOldRice == crispy) {
    whiskedEggs(3)
    soyMix()
    veg()
    _sesameOil(1tsp)
}
