#!/bin/gastro

dutchOven(mediumLowHeat) {
    add: _baconFat
    add: _butter
    render
    add: _onion(2medium){minced}
    add: _cumin(3tsp)
    add: _paprika(2tbsp)
    add: _fennelSeeds(12)
    add: _garlicCloves(4){crushed}
    cook: tillSoft
}

smallBowl() {
    _stewBeef(1.5lbs)
    _flour
    toss
}

dutchOven(mediumLowHeat) {
    add: smallBowl()
    add: _redWine
    add: _cannedTomatoes(8oz)
    add: _beefBroth(2cups)
    add: _bayLeaves
    # Optional:
    add: _carrots(1.5cups){sliced}
    add: _potatoes(3cups){biteSized}
}

# 90 mins cook time, lower the better
# Serve with fresh bread

garnish: _dill, _greenOnion, _sourCream
