#!/bin/gastro

creoleVeg() {
    _onion(1){chopped}
    _celery(2){chopped}
    _redPepper(1){chopped}
}

spices() {
    _oregano
    _hysSeasoning
    _accentMSG
    _fennel(24seeds)
    _paprika
    _cayenne
}

dutchOven(mediumHeat) {
    add: _butter(1cup)
    add: _flour(1cup)
    # Looking for the color of an old penny
    while (color != dark) {
        stir
        wait: 2mins
    }
    add: creoleVeg()
    add: _garlic(6cloves){crushed}
    add: spices()
    add: _sausages(3){skinless}
    add: _chickenThighs(2){skinless, boneless}
    add: _cannedTomatoes(14oz)
    add: _chickenBroth(1quart)
    add: _beefBroth(1quart)
    add: _bayLeaves(3)
    wait: 30-60mins
}

dutchOven(lowHeat, lidOn) {
    wait: 3hrs
}

dutchOven(noHeat) {
    add: _preCookedPrawn
}

# Serve with bread as soup
serve
