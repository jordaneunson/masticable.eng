#!/bin/gastro

# 1lb of shortrib should be 3x3-inch pieces, NOT STRIPS.

preHeat(oven(300F))

dutchOven(mediumHighHeat, _oliveOil) {
    sear() {
        season(_shortRib(1lb))
    }
}

bits = slice(_onion, _shallot, _garlicClove(3))

dutchOven(mediumHeat) {
    bits
    while (bits != soft) {
        _wine(1/3cup)
        _soy(2tbsp)
        _broth(1/3cup)
    }
}

dutchOven(oven(300F))

wait(1.25hrs)

eat()
