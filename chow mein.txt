#!/bin/gastro

soyMix() {
    _garlic(3){crushed}
    _ginger(1"){grated}
    _sesameOil(2tbsp)
    _soy(1/3cup)
    _brownSugar(3tbsp)
    # Originally 1/4 cup, I found this way too sweet.
    _honey(2tbsp)
    # Originally 1/3 cup, I found this way too sweet.
    _meatBroth(1cup)
    _cornstarch(2tbsp)
}

prep() {
    _steak(1lb){stripped}
    _broccoli(2cups){florets}
    _carrots(2cups){sliced thin}
}

mediumPot(highHeat) {
    add: _water
    boil
}

wok(highHeat) {
    add: prep()
    cook: 3mins
    add: soyMix()
}

mediumPot(highHeat) {
    add: _chowMeinNoodles
    cook: alDente
}

serve
