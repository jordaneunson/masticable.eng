#!/bin/gastro

dutchOven(mediumHighHeat) {
    add: _oil
    add: _pancetta(150g){cubed}
    fry: tillCrispy
    remove: _pancetta
}

melange() {
    _carrot(2){blitzed finely}
    _celery(2){blitzed finely}
    _onion{blitzed finely}
}

dutchOven(mediumHighHeat) {
    add: _potRoast
    brown: allSides
    remove: _potRoast
    add: melange()
    cook: tillSoftened
}

dutchOven(highHeat) {
    add: _pancetta
    add: _cannedTomatoes(14oz)
    add: _garlic
    add: _rosemary
    add: _oregano
    mix
    add: _potRoast
}

dutchOven(oven, 300F) {
    wait: 3hrs
}

serve
