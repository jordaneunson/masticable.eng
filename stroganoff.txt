#!/bin/gastro

prepCounter() {
    _steak
    # wash and trim
    season: _steak
}

largeSkillet(mediumHeat) {
    add: _oliveOil
    add: _butter(2tbsp)
    add: _mushrooms(300g){sliced}
    add: _onion(1large){diced}
    add: _garlic(2cloves)
    add: _seasonings
    cook: tillSoft
}

smallPot(lowHeat) {
    add: _butter(2tbsp)
    add: _cream(1/4cup)
    add: _sourCream(150ml)
    add: _flour(1tbsp)
    add: _cornstarch(to thicken as needed)
    stir: untilSmooth
}

largeSkillet(highHeat) {
    add: _beefBroth(2cups)
    add: _bayLeaf
    # bring to boil then lower heat to simmer
    simmer
}

mediumPot(highHeat) {
    add: _water
    boil
    add: _eggNoodles(enough)
    cook: alDente
}

largeSkillet(lowHeat) {
    add: smallPot()
    add: _canPeas(1/2)
    mix
}

stoneSkillet(highHeat) {
    add: _oliveOil
    add: _steak
    wait: 30seconds
    flip
    add: _butter
    add: _thyme
    add: _garlicClove
    while (steak != done) {
        wait: 30seconds
        flip
    }
}

prepCounter() {
    rest: _steak
    slice
}

serve
