#!/bin/gastro

sauce() {
    _feta(4oz)
    _garlic(8cloves)
    _greekYogurt(1/2cup){low fat}
    _water(splash)
    _salt(big pinch)
    _oliveOil(1tbsp)
    blend
}

veg() {
    _cucumber(1/2){sliced thin}
    _redOnion(1/2){sliced thin}
    _freshOregano(1tsp)
    _freshParsley(1tsp)
    _jalapeno(1)
    _salt
    _redWineVinegar(1tbsp)
}

rub() {
    _pepper
    _paprika
    _coriander
    _cumin
    _salt
}

prepCounter() {
    _steak(1)
    coat: rub()
}

castIron(highHeat) {
    add: _oil(1tbsp)
    add: _steak
    cook: tillDone
}

plate {
    add: _pita || _tacos
    add: _steak{sliced}
    add: veg()
    add: sauce()
}
