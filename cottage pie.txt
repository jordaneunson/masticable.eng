#!/bin/gastro

oven(400F) {
    preheat
}

fryingPan(highHeat) {
    add: _groundBeef(340g)
    add: _whiteMushrooms(5-6)
    brown
}

fryingPan(mediumHeat) {
    deglaze: _guinness
    add: _carrot(1/2){minced}
    add: _onion(1/2){minced}
    add: _garlic(5cloves){crushed}
    add: _rosemary
    add: _thyme
    add: _worcestershireSauce
    add: _chickenBroth(1cup)
    simmer: lowHeat
}

largePot(highHeat) {
    add: _russetPotatoes(4){peeled}
    boil: tillSoft
    mash
    add: _butter
    add: _milk
    add: _parm
}

casseroleDish() {
    layer: _fryingPan
    layer: _largePot
}

oven(400F) {
    bake: tillGolden
}

serve
