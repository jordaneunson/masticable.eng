#!/bin/gastro

prepCounter() {
    # Pick as many leaves as you can, more than you think.
    _thymeLeaves
    _rosemary(1/2)
    _broccoliFlorets
    _bayLeaf(2)
}

dutchOven(highHeat) {
    add: _oliveOil
    add: _whiteOnion(1/2){blitzed}, _celeryStalk{blitzed}
    # add _spicyHotSausage(1) for brocencheese version
    fry: tillSoft
    add: _arborioRice
    add: _pinotGris(3/4cup)
    stir
}

dutchOven(mediumHeat) {
    add: prepCounter()
    while (risotto != done) {
        add: _veggieBroth
        stir
    }
}

# at about 80%:
dutchOven(noHeat) {
    add: _butter(1/3cup)
    add: _pecorino(1/2cup)
    add: _cheddarCheese(1/2cup)
    add: _dicedParsley(handful){very fine}
    add: _squeezedLemon(1/2)
    add: _seasoning
    lid
    wait: 10mins
}

serve
