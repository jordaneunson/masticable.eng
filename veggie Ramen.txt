#!/bin/gastro
# yumyum veggie ramen
# inspired by porkBun(), tested on theDread()

smallStockPot(lowHeat) {
    add: _campbellsPhoBroth(quart)
    simmer
}

smallFryPan(mediumHeat) {
    add: _oliveOil
    add: _ginger
    add: _garlic
    add: _leek(1/2){trimmed, quartered, cleaned}
    add: _mushrooms(to match){torn}
    fry: tillSoft
}

smallStockPot(lowHeat) {
    add: smallFryPan()
    stir
    add: _ramenNoodles(2packets)
    add: _eggs(2)
    add: _broccoliCrowns(to match)
}

serve
