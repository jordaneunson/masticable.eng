#!/bin/gastro

dressing() {
    _baconDrippings(2tbsp)
    _dijonMustard(2tbsp)
    _appleCiderVinegar(3tbsp)
    _oliveOil(2tbsp)
    _salt(1tsp)
    _blackPepper(1/2tsp)
}

prepCounter() {
    _potatoes(1kg){cubed, boiled until tender}
    _bacon(6slices){cooked, crumbled}
    _redOnion(1/2){diced}
}

largeBowl() {
    add: prepCounter()
    add: dressing()
    add: _freshParsley(2tbsp){chopped}
    toss
}

serve
