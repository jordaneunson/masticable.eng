#!/bin/gastro

soyMix() {
    _garlic(4){crushed}
    _ginger(2"){grated}
    _sesameOil(2tbsp)
    _soy(1/2cup)
    _riceVinegar(2tbsp)
    _chickenStock(1/2cup)
    _cornstarch(2tbsp)
}

prep() {
    _steak(1lb){stripped}
    _broccoli(2cups){florets}
    _carrots(2cups){sliced thin}
    _redPepper(0.5cup)
    _redOnion(0.5cup)
    _celery(0.5cup)
    _redChillies(3){de-seeded}
}

mediumPot(highHeat) {
    add: _water
    boil
}

wok(highHeat) {
    add: prep()
    cook: 3mins
    add: soyMix()
}

mediumPot(highHeat) {
    add: _chowMeinNoodles
    cook: alDente
}

garnish: _cilantro, _peanuts
