#!/bin/gastro

dressing() {
    _mayonnaise(1/2cup)
    _whiteWineVinegar(3tbsp)
    _appleCiderVinegar(2tbsp)
    _canolaOil(2tbsp)
    _honey(2tbsp)
    _sugar(1tbsp)
    _salt(1/2tsp)
    _blackPepper(to taste)
    _celerySeed(1/2tsp)
    _mustard(1tsp){Dijon}
}

prepCounter() {
    _cabbage(4cups){shredded}
    _carrot(1){shredded}
    _redOnion(1/4){sliced thin}
}

largeBowl() {
    add: prepCounter()
    add: dressing()
    toss
}

serve
