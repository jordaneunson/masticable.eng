#!/bin/gastro

oven(350F) {
    preheat
}

fryPan(mediumHeat) {
    add: _onion(1/2){diced}
    add: _mushrooms{sliced}
    add: _truffle(as much as u can afford)
    add: _butter(1/3cup)
    cook: tillSoft
    add: _cream(1/2litre)
    add: _cornstarch
    add: _seasoning
    stir: untilThickened
}

bakePan(oven, 350F) {
    layer: _sauce
    add: _cannelonis
    cover: _sauce
    bake: 20mins
}

serve
