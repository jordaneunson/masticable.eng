dutchOven(mediumHeat) {
    roux() {
        _butter(1cup)
        _flour(1cup)
        while(color != dark) {
            stir()
            wait(2mins)
        }
        # Looking for the color of an old penny
    }
    
    creoleVeg() {
        _onion(1){chopped}
        _celery(2){chopped}
        _redPepper(1){chopped}
    }
    
    _creoleVeg()
    _garlic(6cloves){crushed}
    
    _spices() {
        _oregano
        _hysSeasoning
        _accentMSG
        _fennel(24seeds)
        _paprika
        _cayenne
    }
    
    _sausages(3){skinless}
    _chickenThighs(2){skinless, boneless} # Optional
    _cannedTomatoes(14oz)
    _chickenBroth(1quart)
    _beefBroth(1quart)
    _bayLeaves(3)
    
    wait(30-60mins)
}
