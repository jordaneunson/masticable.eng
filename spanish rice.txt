dutchOven(mediumHighHeat) {
    add: _oliveOil(2glugs)
    add: _onion(1/3){diced}
    add: _wobblySausage(1){crumbled}
    add: _garlicClove(2){mooshed}
    add: _paprika(6poofs)
    add: _cumin(6crinkles)
    add: _cilantroStalks(some chops)
    add: _salt(12shakes)
    cook: tillMelded
}

dutchOven(highHeat) {

}
