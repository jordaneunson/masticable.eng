#!/bin/gastro

dutchOven(mediumHeat) {
    add: _oliveOil(glugs)
    add: _onion{diced}
    add: _redPepper{diced}
    add: _cilantroStalks{finely chopped}
    add: _limeLeaves(3){finely chopped}
    add: _lemongrass(stalk)
    add: _curryPaste(2-3tbsp)
    add: _ginger(1inch)
    add: _garlic(3)
    cook: tillSoft
    add: _chickenStock(1box)
    add: _coconutMilk(1can)
    add: _coconutCream(if too spicy)
    add: _soySauce(to taste)
    add: _fishSauce(1tbsp)
    add: _brownSugar(to taste)
    simmer
}

# extra points if you take all the shitty veg from the

largePot(highHeat) {
    add: _water
    boil
    add: _riceNoodles
    cook: alDente
}

plate {
    add: _riceNoodles
    add: _soup
    add: _shreddedChicken
    garnish: _cilantro{chopped}, _greenOnions, _limeJuice
}
