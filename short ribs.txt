#!/bin/gastro
# 1lb of shortrib should be 3x3-inch pieces, NOT STRIPS.

oven(300F) {
    preheat
}

dutchOven(mediumHighHeat) {
    add: _oliveOil
    add: _shortRib(1lb){seasoned}
    sear: allSides
    remove: _shortRib
}

dutchOven(mediumHeat) {
    add: _onion{sliced}
    add: _shallot{sliced}
    add: _garlicClove(3){sliced}
    cook: tillSoft
    add: _wine(1/4cup)
    add: _soy(2tbsp)
    add: _broth(1/4cup)
    add: _shortRib
}

dutchOven(oven, 300F) {
    wait: 3hrs
}

serve
